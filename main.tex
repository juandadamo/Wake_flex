\begin{eqnarray}
F_D  = \rho \int {U_\infty (1-u(x)) }dy +\int {\rho \overline{u_x^\prime u_x^\prime} }dy - \int \bar p dy
\end{eqnarray}
\begin{eqnarray}
-\frac{1}{\rho}\frac{\partial \bar p}{\partial y} = \left(\bar u_x \frac{\partial \bar u_y}{\partial x}  + \bar u_y  \frac{\partial \bar u_y}{\partial y}  \right)+\left( \frac{\partial \overline{u_x^\prime u_y^\prime }}{\partial x} + \frac{\partial \overline{u_y^\prime u_y^\prime }}{\partial y}    \right)-\nu\left(\frac{\partial ^2 \bar u_y}{\partial^2 x} + \frac{\partial ^2 \bar u_y}{\partial^2 y}\right)
\end{eqnarray}
Where $\bar{()}$ is time average, and $()^\prime$ stands for flow fluctuations.
